function sort(arr) {
    for (let i = 0; i < arr.length; i++) {
        let minIndex = 0
        for (let j = i; j < arr.length; j++) {
            if(arr[j] < arr[minIndex])
                minIndex = j
        }
        swap(arr,i,minIndex)
    }
}

function swap(arr,index,minIndex) {
    const t = arr[index]
    arr[index] = arr[minIndex]
    arr[minIndex] = t
}
